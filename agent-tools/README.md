```python

task = BrowserTask(
    goal="Extract pricing from page",
    url="https://example.com/pricing",
    actions=[
        {"type": "wait", "seconds": 2},
        {"type": "scroll", "direction": "down"},
        {"type": "extract", "selector": ".price"}
    ]
)
```
